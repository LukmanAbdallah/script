#!/bin/bash
echo "Check disk Usage"
disk_usage=$(df -h | grep '/dev/sdc' | awk '{print $5}' | cut -d '%' -f 1)
echo "Your disk is filled with $disk_usage%"
if [ "$disk_usage" -gt 80 ]; then
    echo "Your disk is almost full. Please consider removing some unwanted things from your machine or perform a disk cleanup."
else
    echo "There is available space on it."
fi
